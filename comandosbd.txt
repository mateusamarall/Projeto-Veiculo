create database projeto_db;
create table estados ( 
sigla varchar(4) primary key, 
nome varchar(25) not null);

create table marca_mod (
marca_mod varchar(25) primary key);

create table lojas(
nome_l varchar(25) not null ,
loja_id varchar(25) primary key);

create table user ( 
login varchar(25) primary key,
nome varchar(25) not null,
senha int not null);  

create table veiculos (
placa varchar(25) primary key,
preco float(7) not null,
ano int not null,
estado varchar(25) ,
marca_mod varchar(25),
loja_id varchar(25) ,
foreign key  (estado) references  estados(sigla),
foreign key (marca_mod) references  marca_mod(marca_mod),
foreign key (loja_id) references lojas(loja_id));
